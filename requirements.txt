#   Ubuntu/Debian: sudo apt-get install -y python3-gi gir1.2-webkit2-4.1 libgirepository1.0-dev
#   Fedora:        sudo dnf install -y python3-gobject webkit2gtk4.1 gobject-introspection-devel
